    <mx:VBox width="100%" height="100%" paddingLeft="50" paddingTop="25">        
        <mx:Text htmlText="Flex ImageCropper: &lt;br /&gt;&lt;br /&gt;This example shows you how to use the following classes: &lt;br /&gt; - flash.net.FileReference for uploading an image &lt;br /&gt;- be.nascom.flash.net.upload.ImageUploader for saving the BitmapData&lt;br /&gt;&lt;br /&gt;"/>        
        <components:ImageCropper id="imageCropper" />        
        <mx:HBox>
            <mx:Button label="Upload" click="uploadImage(event)"/>
            <mx:Button label="Save" click="saveImage(event)"/>    
        </mx:HBox>        
        <mx:Text id="debugText" text="See View Source for the details" width="100%" height="100%"/>            
    </mx:VBox>
    
    <mx:Script>
        <![CDATA[
            import be.nascom.flash.events.FileEvent;
            import be.nascom.flash.net.upload.ImageUploader;
            import mx.utils.ObjectUtil;
            import mx.controls.Alert;
            
            private const UPLOAD_PATH:String = "backend/upload.php";
        
            private var urlRequest:URLRequest;
            private var fileReference:FileReference;
        
            private function uploadImage(event:Event):void {
                
                //Add extra params here is necessary
                var postparams:URLVariables = new URLVariables();                
                //postparams.param =  value;
                                  
                urlRequest = new URLRequest();
                urlRequest.url = UPLOAD_PATH;
                urlRequest.data = postparams;
                
                try {
                    fileReference=new FileReference();
                    fileReference.browse(getImageTypes());
                    fileReference.addEventListener(Event.SELECT,handleSelect);
                    fileReference.addEventListener(IOErrorEvent.IO_ERROR, handleIOError);
                    fileReference.addEventListener(DataEvent.UPLOAD_COMPLETE_DATA, handleUploadComplete);
                } catch (illegalOperation:IllegalOperationError) {
                    debugText.text = "illegalOperation";
                }    
            }
            
            protected function getImageTypes():Array {
                var allTypes:Array = new Array();
                allTypes.push(new FileFilter("Images (*.jpg, *.jpeg, *.gif, *.png)","*.jpg;*.jpeg;*.gif;*.png"));
                return allTypes;
            }            
                        
            private function handleSelect(event:Event):void {
                fileReference.upload(urlRequest);
            }
            
            private function handleUploadComplete(event:DataEvent):void {
                debugText.text = ObjectUtil.toString(event);
                imageCropper.source = "backend/" + event.data;
            }
            
            protected function handleIOError(event:IOErrorEvent):void{
                debugText.text = "IO error has occured: " + ObjectUtil.toString(event);
            }
            
            private function saveImage(event:Event):void {
                var uploader:ImageUploader = new ImageUploader(UPLOAD_PATH);
                uploader.upload(imageCropper.croppedBitmapData, "cropped_image.jpg", 100, "jpg", "Filedata");
                uploader.addEventListener(FileEvent.FILE_UPLOADED, handleSaveComplete);
                uploader.addEventListener(IOErrorEvent.IO_ERROR, handleIOError);    
            }
            
            private function handleSaveComplete(event:FileEvent):void {
                debugText.text = ObjectUtil.toString(event) 
            }
        ]]>
    </mx:Script>
    
</mx:Application>
